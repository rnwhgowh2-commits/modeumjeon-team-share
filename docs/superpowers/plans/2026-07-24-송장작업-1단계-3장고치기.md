그리고 `function rowPass(r,skipCol){` (643줄) **바로 위**에 헬퍼를 추가한다:

```js
    // 찾기 대상 = 그 행이 가진 **모든 값**(상품명·수령자만이 아니다 — 송장번호·주문번호·
    //  계정·주소로도 찾을 수 있어야 한다). 값이 많아 행마다 한 번만 만들어 캐시한다.
    //  _ 로 시작하는 키는 화면에 없는 내부값이라 뺀다(엉뚱한 것에 걸리지 않게).
    function srchHay(r){
      if(r.__hay!==undefined)return r.__hay;
      var parts=[];
      for(var k in r){
        if(!Object.prototype.hasOwnProperty.call(r,k))continue;
        if(k.charAt(0)==='_')continue;
        var v=r[k];
        if(v===null||v===undefined||typeof v==='object')continue;
        parts.push(String(v));
      }
      try{Object.defineProperty(r,'__hay',{value:parts.join(' ').toLowerCase(),enumerable:false});}
      catch(e){r.__hay=parts.join(' ').toLowerCase();}
      return r.__hay;
    }
```
