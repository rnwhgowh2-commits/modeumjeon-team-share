# C·D · 소싱처별 혜택 설정 탭 (베이스 편집) — 구현 플랜

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development (recommended) or executing-plans. Steps use `- [ ]` checkboxes.

**Goal:** 가이드 지도에 **"소싱처별 혜택 설정"** 탭(8번째)을 추가해 소싱처별 혜택 베이스(이름·%·유형)를 추가/수정/삭제하고, 저장 시 기존 동기화·따라쓰기 배선으로 전 모음전·최종매입가에 반영. 편집 장소를 **딱 2곳**(① 이 탭=베이스 / ② 모음전 영수증 카드=상품별)으로 정리.

**Architecture:** 백엔드는 **전부 재사용** — `PUT /sourcing-guide/api/<sid>`(→ `validate_guide` → `sync_templates_from_crawl_guide` → 선택적 `snapshot_bundle_from_templates`). 새 탭은 JS로 소싱처 선택 시 `GET /sourcing-guide/api/<sid>`로 혜택을 불러와 카드로 편집하고 `PUT`으로 저장(detail.js `collect()`와 동일 페이로드). 옛 `/sourcing-guide/<sid>` 상세의 ③ 혜택 편집기는 제거(읽기 안내로 대체). 상품별 편집·**혜택 초기화는 영수증 카드(매트릭스 fx 팝오버 `smBundleReset`)에 이미 존재** → 변경 없음, 검증만.

**Tech Stack:** Flask(Jinja2), 순수 HTML/CSS/JS, pytest(라우트·템플릿 내용), 라이브 검증.

**기준 브랜치:** origin/main 분기 워크트리 `C:/dev/_wt_cd` (branch feature/source-benefit-tab).

**value_mode 없음** (사용자 결정 2026-06-23): 베이스=고정 혜택(수기 관리), 크롤 동적 혜택은 `compute_breakdown`이 자동으로 위에 얹음(현행 유지). ⚠️ 같은 혜택을 베이스+크롤 중복 시 이중차감 — 탭에 경고 문구 1줄.

**조건부 게이트(키워드 포함/제외) 연기** (사용자 결정 2026-06-23): 혜택별 조건부 규칙은 **별도 전용 단계**. 이유 = `gate_benefits`(triggers/excludes 게이트)가 현재 **미리보기 엔드포인트(`/api/<sid>/gate-preview`)에서만** 호출되고 실제 크롤·`compute_breakdown`에는 미배선 → 조건을 정의해도 최종가에 안 먹음. 실배선(+혜택별 excludes 신규)은 금전 핵심부(compute_breakdown, 하드코딩된 SSG MONEY 등) 수술이라 전용 플랜+검증으로 분리. **본 C·D는 이름·값·유형(최종가에 바로 반영되는 부분)만.** ⚠️ 그래도 저장 시 기존 benefit의 triggers/match 등 조건부 필드는 **보존**(collectBenefits가 원본 merge) — 향후 조건부 단계에서 재사용.

**범위 밖:** value_mode, 재고 설정, 매트릭스 영수증 카드 신규 기능(이미 존재).

<script>
(function(){
  var sel=document.getElementById('bs-source'); if(!sel) return;
  var list=document.getElementById('bs-list'), status=document.getElementById('bs-status');
  var saveBtn=document.getElementById('bs-save'), addBtn=document.getElementById('bs-add');
  var applyCk=document.getElementById('bs-apply-ck'), applyBtn=document.getElementById('bs-apply');
  var curGuide=null;   // 불러온 전체 guide (benefits만 교체 후 PUT)
  var TYPES=['정률(%)','정액(원)','적립(%→원)','고정액'];

  var CURB=[];   // 현재 소싱처 benefits 원본 (조건부 필드 보존용)
  function cardHTML(b,idx){
    b=b||{};
    var method=b.method||'정률(%)';
    var opts=TYPES.map(function(t){return '<option'+(t===method?' selected':'')+'>'+t+'</option>';}).join('');
    var unit=(method.indexOf('%')>=0)?'%':'원';
    return '<div class="bcard" data-idx="'+(idx==null?'':idx)+'">'
      +'<input class="bn" value="'+(b.name||'').replace(/"/g,'&quot;')+'" placeholder="혜택명">'
      +'<input class="bval" inputmode="decimal" value="'+(b.value==null?'':b.value)+'" title="'+unit+'">'
      +'<select class="btype">'+opts+'</select>'
      +'<button type="button" class="bdel">×</button></div>';
  }
  function renderBenefits(benefits){
    CURB=benefits||[];
    list.innerHTML=CURB.map(function(b,i){return cardHTML(b,i);}).join('') || '<div class="bs-status">혜택 없음 — [+ 혜택 추가]</div>';
  }
  function collectBenefits(){
    return [].map.call(list.querySelectorAll('.bcard'), function(c){
      var name=c.querySelector('.bn').value.trim();
      var v=c.querySelector('.bval').value.trim();
      var method=c.querySelector('.btype').value;
      var idx=c.dataset.idx;
      var orig=(idx!=='' && idx!=null)?(CURB[+idx]||{}):{};
      // ⚠️ 원본 benefit 보존 + 편집칸(name/value/method)만 덮어씀.
      //    triggers/match/base/freq/status 등 조건부 필드를 파괴하지 않음(별도 조건부 단계 대비).
      return Object.assign({}, orig, {name:name, value:(v===''?null:parseFloat(v)), method:method,
              apply:orig.apply||'deduct', status:orig.status||'always'});
    }).filter(function(b){return b.name;});
  }
  function load(sid){ status.textContent='불러오는 중…';
    fetch('/sourcing-guide/api/'+sid).then(function(r){return r.json();}).then(function(j){
      if(!j.ok){status.textContent='불러오기 실패';return;}
      curGuide=j.guide; renderBenefits((j.guide.pricing||{}).benefits); status.textContent='';
    }).catch(function(){status.textContent='불러오기 오류';});
  }
  function save(applyBundles){
    if(!curGuide){return;}
    var g=JSON.parse(JSON.stringify(curGuide));
    g.pricing=g.pricing||{}; g.pricing.benefits=collectBenefits();
    var payload=Object.assign({},g); if(applyBundles) payload.apply_to_bundles=true;
    status.textContent='저장 중…'; saveBtn.disabled=true;
    fetch('/sourcing-guide/api/'+sel.value,{method:'PUT',headers:{'Content-Type':'application/json'},
      body:JSON.stringify(payload)}).then(function(r){return r.json();}).then(function(j){
      saveBtn.disabled=false;
      if(!j.ok){status.textContent='저장 실패: '+(j.message||j.error||'');return;}
      curGuide=j.guide; renderBenefits((j.guide.pricing||{}).benefits);
      status.textContent='저장됨'+(applyBundles?' · 따라쓰기 '+(j.bundles_applied||0)+'개':' (동기화 '+(j.benefits_synced||0)+')');
    }).catch(function(){saveBtn.disabled=false;status.textContent='저장 오류';});
  }
  sel.addEventListener('change',function(){load(sel.value);});
  addBtn.addEventListener('click',function(){ list.insertAdjacentHTML('beforeend',cardHTML(null)); });
  list.addEventListener('click',function(e){ if(e.target.classList.contains('bdel')) e.target.closest('.bcard').remove(); });
  list.addEventListener('change',function(e){ if(e.target.classList.contains('btype')){ var c=e.target.closest('.bcard'); c.querySelector('.bval').title=(e.target.value.indexOf('%')>=0?'%':'원'); }});
  saveBtn.addEventListener('click',function(){ save(false); });
  applyCk.addEventListener('change',function(){ applyBtn.disabled=!applyCk.checked; });
  applyBtn.addEventListener('click',function(){ if(applyCk.checked && confirm('모든 모음전이 이 소싱처 기본값으로 덮어써집니다. 진행할까요?')) save(true); });
  if(sel.options.length) load(sel.value);   // 초기 로드
})();
</script>
