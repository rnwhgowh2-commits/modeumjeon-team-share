> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 크롤할 때마다 소싱처 URL별로 "재고 변동 / 가격 변동" 여부를 자동으로 감지·기록하고, 계수(1~5)·무변동 연속 횟수를 저장할 데이터 토대를 만든다.

```python
"""소싱처 URL 1건의 직전 옵션값 vs 새 옵션값 비교 → 변동 여부 판정 (순수 함수).

키 = (color_text, size_text). 매칭되는 옵션의 재고·가격이 바뀌었는지,
옵션이 새로 생기거나 사라졌는지를 본다. 폴백 없음 — 값 그대로 비교.
"""
from typing import Any


def _key(o: dict) -> tuple:
    return (o.get("color_text"), o.get("size_text"))


def detect_changes(old_options: list[dict], new_options: list[dict]) -> dict[str, Any]:
    """반환: {'stock_changed': bool, 'price_changed': bool, 'detail': str}."""
    old_map = {_key(o): o for o in old_options}
    new_map = {_key(o): o for o in new_options}

    stock_changed = False
    price_changed = False
    notes: list[str] = []

    # 옵션 등장/소멸 = 재고 상태 변화
    added = set(new_map) - set(old_map)
    removed = set(old_map) - set(new_map)
    if added or removed:
        stock_changed = True
        for k in added:
            notes.append(f"[{k[0]}/{k[1]}] 옵션 생김")
        for k in removed:
            notes.append(f"[{k[0]}/{k[1]}] 옵션 사라짐")

    # 공통 옵션의 재고·가격 비교
    for k in set(old_map) & set(new_map):
        o, n = old_map[k], new_map[k]
        if o.get("stock") != n.get("stock"):
            stock_changed = True
            notes.append(f"[{k[0]}/{k[1]}] 재고 {o.get('stock')}→{n.get('stock')}")
        if o.get("price") != n.get("price"):
            price_changed = True
            notes.append(f"[{k[0]}/{k[1]}] 가격 {o.get('price')}→{n.get('price')}")

    return {
        "stock_changed": stock_changed,
        "price_changed": price_changed,
        "detail": " · ".join(notes),
    }
```

```python
from lemouton.sources.models import SourceOption, CrawlDelta


def _mk_product(db):
    sp = SourceProduct(site="musinsa", url="https://x/9")
    db.add(sp); db.flush()
    return sp


def test_first_crawl_records_delta_and_streak(db):
    sp = _mk_product(db)
    opts = [{"color_text": "블랙", "size_text": "220", "price": 50000, "stock": 3}]
    persist_crawled_options(db, source_product=sp, options=opts)

            .order_by(CrawlDelta.id.desc()).first())
    assert last.stock_changed is False and last.price_changed is False
    assert sp.no_change_streak == 1  # 무변동 1회 누적


def test_change_resets_streak(db):
    sp = _mk_product(db)
    a = [{"color_text": "블랙", "size_text": "220", "price": 50000, "stock": 3}]
    b = [{"color_text": "블랙", "size_text": "220", "price": 50000, "stock": 0}]
    persist_crawled_options(db, source_product=sp, options=a); db.flush()
    persist_crawled_options(db, source_product=sp, options=a); db.flush()  # streak=1
    persist_crawled_options(db, source_product=sp, options=b); db.flush()  # 재고변동
```

```python
def changed_product_ids_since(session, *, only_latest: bool = True) -> set[int]:
    """변동(재고 또는 가격)이 있는 source_product_id 집합.

    only_latest=True 면 각 URL의 '가장 최근 CrawlDelta' 만 보고 판단
    (업로드 게이트용 — 지금 보낼지 말지). False 면 하나라도 변동이면 포함.
    """
    from lemouton.sources.models import CrawlDelta
    rows = (session.query(CrawlDelta)
            .order_by(CrawlDelta.source_product_id, CrawlDelta.id.desc())
            .all())
    result: set[int] = set()
    seen: set[int] = set()
    for r in rows:
        if only_latest:
            if r.source_product_id in seen:
                continue
            seen.add(r.source_product_id)
        if r.stock_changed or r.price_changed:
            result.add(r.source_product_id)
    return result
```
