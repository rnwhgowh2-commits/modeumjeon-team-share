# 소싱처 컬럼 (소싱처×URL) 단위 재정의 — 구현 플랜

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development 로 태스크 단위 구현. 각 스텝 체크박스(`- [ ]`).

**Goal:** 매트릭스·매트릭스보기·크롤로그의 소싱처 컬럼을 "등록 소싱처 고정 7개"에서 "이 모음전에 URL이 등록된 (소싱처×URL) 조합"으로 재정의 — URL 없는 소싱처 컬럼 숨김(F1) + 다중 URL 별개 컬럼 분리(F2).

**Architecture:** 백엔드는 `bundle_source_url_id` 1개만 추가 노출. 프론트가 `deriveSourceColumns(DATA)` 순수 헬퍼로 (source_id, URL) 컬럼 목록을 파생해 매트릭스·팝업·로그에 동일 적용. 가격/혜택/매칭 계산식 불변 — "표시 라우팅"만 변경.

**Tech Stack:** Flask/SQLAlchemy(`api_pricing.py`), Vanilla JS(`_matrix_v3.html`, `crawl_log.js`), pytest. 작업 워크트리 `C:/dev/_wt_cols`(branch `feature/source-url-columns`), app root `프로그램/_시스템`.

**스펙**: `docs/superpowers/specs/2026-06-23-소싱처컬럼-URL단위재정의-design.md`

- [ ] **Step 1: 헬퍼 추가** — IIFE 내부(`DATA` 접근 가능 지점, `loadMatrix` 부근)에 순수 함수:

```js
// 소싱처 컬럼 = (source_id × URL) 파생. 0URL 소싱처 제외(F1), 다중URL 분리(F2).
function deriveSourceColumns(DATA){
  const opts = (DATA && DATA.options) || [];
  const reg = (DATA && DATA.sources) || [];           // 정렬·이름 참고용
  const orderOf = {}; reg.forEach((s,i)=>{ orderOf[s.id] = (s.sort_order!=null?s.sort_order:i); });
  // 1) (source_id,colKey) 수집 + 메타
  const seen = new Map();  // key `${sid}|${colKey}` -> {source_id, colKey, source_name, bsu_id, url_label, url_sort}
  opts.forEach(o => (o.sources||[]).forEach(s => {
    if (s.source_id == null) return;
    const colKey = (s.bundle_source_url_id != null) ? ('b'+s.bundle_source_url_id) : ('u'+(s.product_url||''));
    const k = s.source_id + '|' + colKey;
    if (!seen.has(k)) seen.set(k, {
      source_id: s.source_id, colKey,
      source_name: s.source_name || s.name || '',
      bsu_id: (s.bundle_source_url_id!=null? s.bundle_source_url_id : null),
      url_label: s.label || s.url_type || '',
      url_sort: (s.sort_order!=null? s.sort_order : (s.bundle_source_url_id!=null? s.bundle_source_url_id : 0)),
    });
  }));
  // 2) 소싱처별 그룹 → URL 정렬 → 번호/라벨
  const bySrc = {};
  seen.forEach(c => { (bySrc[c.source_id] = bySrc[c.source_id]||[]).push(c); });
  const cols = [];
  Object.keys(bySrc)
    .sort((a,b)=> (orderOf[a]??999)-(orderOf[b]??999) || (a-b))
    .forEach(sid => {
      const arr = bySrc[sid].sort((x,y)=> (x.url_sort-y.url_sort) || String(x.colKey).localeCompare(String(y.colKey)));
      const total = arr.length;
      arr.forEach((c,i)=>{
        c.idx = i+1; c.total = total;
        c.name = (total<=1) ? c.source_name
                            : (c.source_name + '(' + (i+1) + ')' + (c.url_label? ' '+c.url_label : ''));
        cols.push(c);
      });
    });
  return cols;  // [{source_id, colKey, name, source_name, bsu_id, url_label, idx, total}]
}
window.deriveSourceColumns = deriveSourceColumns;  // 팝업/테스트 접근
```

- [ ] **Step 2: 셀 매칭 헬퍼** — 한 컬럼에 해당하는 소싱처 항목 필터(같은 source_id & 같은 colKey):

```js
function _srcEntriesForCol(o, col){
  return (o.sources||[]).filter(x =>
    x.source_id === col.source_id &&
    (((x.bundle_source_url_id!=null)?('b'+x.bundle_source_url_id):('u'+(x.product_url||''))) === col.colKey)
  );
}
```
