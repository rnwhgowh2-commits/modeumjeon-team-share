```python
"""더망고 매입 엑셀 파서."""
import io

import pandas as pd
import pytest

from lemouton.margin import buy_parser as bp


def _xlsx_bytes(rows: list, cols: list) -> bytes:
    buf = io.BytesIO()
    pd.DataFrame(rows, columns=cols).to_excel(buf, index=False, engine="openpyxl")
    return buf.getvalue()


COLS = ["마켓주문일자", "마켓명", "마켓주문번호", "수령인명",
        "마켓상품명", "옵션1", "사이트주문번호", "구매가격", "간단메모"]


def test_parse_buy_basic():
    data = _xlsx_bytes([
        ["26.07.04", "쿠팡", "1001", "홍길동", "코트 12345", "블랙/95", "SO-1", 50000, ""],
    ], COLS)
    df = bp.parse_buy(data, "더망고.xlsx")
    assert len(df) == 1
    assert df.loc[0, "마켓주문번호"] == "1001"
    assert df.loc[0, "구매가격"] == 50000
    assert "_uid" in df.columns


def test_parse_buy_strips_float_suffix_on_order_no():
    data = _xlsx_bytes([
        ["26.07.04", "쿠팡", 1001.0, "홍길동", "코트 12345", "블랙/95", "SO-1", 50000, ""],
    ], COLS)
    df = bp.parse_buy(data, "더망고.xlsx")
    assert df.loc[0, "마켓주문번호"] == "1001"


def test_parse_buy_sentinel_999_becomes_zero():
    """구매가격 미입력 센티널 999999999.99 는 0 으로 — 집계 왜곡 방지."""
    data = _xlsx_bytes([
        ["26.07.04", "쿠팡", "1001", "홍길동", "코트 12345", "블랙/95", "SO-1", 999999999.99, ""],
    ], COLS)
    df = bp.parse_buy(data, "더망고.xlsx")
    assert df.loc[0, "구매가격"] == 0


def test_parse_buy_missing_required_column_raises():
    data = _xlsx_bytes([["26.07.04", "쿠팡"]], ["마켓주문일자", "마켓명"])
```

```python
# -*- coding: utf-8 -*-
"""매출(SellRow) 생산자 2종 — 마켓 API / 샵마인 엑셀.

두 생산자는 동일한 SELL_COLUMNS 스키마를 뱉는다. matcher 는 출처를 모른다.
컬럼명은 기존 샵마인 이름을 그대로 쓴다 — matcher 를 무수정으로 두기 위한 의도적 선택.

정산예정금액 정책의 유일한 자리. 스펙 §4 참조.
"""
import io
import logging
import re

import pandas as pd

from lemouton.margin.config import COUPANG_FEE_RATE

logger = logging.getLogger(__name__)

# matcher 가 읽는 컬럼 + 마진 표시에 필요한 컬럼
SELL_COLUMNS = [
    "오픈마켓주문번호", "상품명", "옵션", "수량", "단가", "실결제금액",
    "정산예상금액_배송비포함", "마켓수수료", "수수료율", "쇼핑몰",
    "수취고객명", "주문일", "송장입력", "주문상태",
    "_settle_source",   # real | estimated | none
    "_sell_origin",     # api | shopmine
]

_SENTINEL_999 = 999999999.99


def _to_numeric_safe(series: pd.Series) -> pd.Series:
    """숫자 컬럼 안전 변환. NaN→0, 999999999.99 센티널→0."""
    result = pd.to_numeric(series, errors="coerce").fillna(0)
    return result.replace(_SENTINEL_999, 0)


def _read_excel_any(file_bytes: bytes, filename: str) -> pd.DataFrame:
    """xls(xlrd) → xlsx(openpyxl) → HTML 형식 xls(html5lib) 순 fallback."""
    attempts = []
    for engine in ("xlrd", "openpyxl"):
        try:
            return pd.read_excel(io.BytesIO(file_bytes), engine=engine)
        except Exception as e:  # noqa: BLE001
            attempts.append(f"{engine}: {e}")

    try:
        text = file_bytes.decode("utf-8")
    except UnicodeDecodeError:
        text = file_bytes.decode("euc-kr", errors="replace")

    if "Excel Workbook Frameset" in text or ("File-List" in text and ".files/" in text):
        m = re.search(r'href\s*=\s*["\']?([^"\'>\s]+\.files)/', text)
        folder = m.group(1) if m else f"{filename.rsplit('.', 1)[0]}.files"
        raise ValueError(
            f'이 파일은 "Excel 웹 페이지" 포맷입니다 — 실제 데이터는 옆의 '
            f'"{folder}" 폴더 안 sheet001.htm 에 있습니다.')

    try:
        dfs = pd.read_html(io.StringIO(text), flavor="html5lib")
        df = max(dfs, key=len)
        if df.iloc[0].astype(str).str.contains("주문|상품|쇼핑몰|단가", regex=True).any():
            df.columns = df.iloc[0]
            df = df.iloc[1:].reset_index(drop=True)
        return df
    except Exception as e:  # noqa: BLE001
        attempts.append(f"pd.read_html: {e}")

    raise ValueError(
        "매출 엑셀 파싱 실패 — 지원되지 않는 형식입니다. 시도한 방식: "
        + " / ".join(f"[{i+1}] {a}" for i, a in enumerate(attempts)))


def from_shopmine_excel(file_bytes: bytes, filename: str) -> pd.DataFrame:
    """샵마인 통합주문관리 엑셀 → SellRow DF.

    쿠팡 '알수없음' 정산금액·수수료는 실결제금액 × (1 − 0.1155) 로 보정한다
    (원본 data_loader.parse_sell 과 동일). 보정값도 샵마인이 그렇게 써 왔으므로
    _settle_source='real' 로 둔다 — 옛 프로그램과의 회귀 동치를 깨지 않기 위함.
    """
    df = _read_excel_any(file_bytes, filename)
    df.columns = [re.sub(r"\s+", " ", str(c)).strip() for c in df.columns]

    col_map = {}
    for c in df.columns:
        s = str(c)
        if "오픈마켓" in s and "주문번호" in s:
            col_map[c] = "오픈마켓주문번호"
        elif "정산예상금액" in s and "배송비" in s:
            col_map[c] = "정산예상금액_배송비포함"
    df = df.rename(columns=col_map)

    if "삼품명" in df.columns and "상품명" not in df.columns:
        df = df.rename(columns={"삼품명": "상품명"})

    # 쿠팡 '알수없음' 보정
    for col in ("정산예상금액_배송비포함", "마켓수수료"):
        if col in df.columns:
            mask = df[col].astype(str).str.contains("알수없음", na=False)
            paid = pd.to_numeric(df.loc[mask, "실결제금액"], errors="coerce")
            if col == "정산예상금액_배송비포함":
                df.loc[mask, col] = paid * (1 - COUPANG_FEE_RATE)
            else:
                df.loc[mask, col] = paid * COUPANG_FEE_RATE
            df[col] = _to_numeric_safe(df[col])

    if "수수료율" in df.columns:
        mask = df["수수료율"].astype(str).str.contains("알수없음", na=False)
        df.loc[mask, "수수료율"] = "11.55%"

    for col in ("단가", "실결제금액"):
        if col in df.columns:
            df[col] = _to_numeric_safe(df[col])
    if "수량" in df.columns:
        df["수량"] = pd.to_numeric(df["수량"], errors="coerce").fillna(1).astype(int)

    df["_settle_source"] = "real"
    df["_sell_origin"] = "shopmine"

    for col in SELL_COLUMNS:
        if col not in df.columns:
            df[col] = ""
    return df
```

```python
# order_export 의 '판매처' 한글값 → 샵마인 '쇼핑몰' 코드값
_PANMAECHEO_TO_SHOPMINE = {
    "스마트스토어": "04.스마트스토어",
    "쿠팡": "06.쿠팡",

    "옥션": "02.옥션",
    "G마켓": "01.지마켓",
}


def market_to_shopmine(panmaecheo: str) -> str:
    """order_export '판매처' → 샵마인 '쇼핑몰'. 미지원 값은 원본 그대로."""
    return _PANMAECHEO_TO_SHOPMINE.get(str(panmaecheo).strip(), str(panmaecheo).strip())


def _to_int_or_blank(v):
    try:
        return int(v)
    except (TypeError, ValueError):
        return ""


def _settlement_for(row: dict):
    """SellRow 의 정산예상금액_배송비포함 + _settle_source 결정. 스펙 §4.

    롯데온만 재계산한다 — order_export 가 정산액 자리에 actualAmt(실결제)를 넣기 때문.
    actualAmt 는 배송비를 이미 포함하므로 배송비를 다시 더하지 않는다.
    """
    src = row.get("_settle_source", "none")
    if row.get("판매처") == "롯데온":
        paid = _to_int_or_blank(row.get("실결제금액"))
        fee = _to_int_or_blank(row.get("마켓수수료"))
        if paid == "" or fee == "" or fee <= 0:
            return "", "none"
        return paid - fee, "real"

    if src == "none":
        return "", "none"
    settle = _to_int_or_blank(row.get("정산예정금(배송비포함)"))
    if settle == "":
        return "", "none"
    return settle, src


def _rows_to_df(rows: list) -> pd.DataFrame:
    """order_export 행 리스트 → SellRow DF."""
    out = []
    for r in rows:
        settle, src = _settlement_for(r)
        out.append({
            "오픈마켓주문번호": str(r.get("오픈마켓주문번호", "") or ""),
            "상품명": r.get("상품명", ""),
            "옵션": r.get("옵션", ""),
            "수량": int(r.get("수량") or 1),
            "단가": _to_int_or_blank(r.get("단가")) or 0,
            "실결제금액": _to_int_or_blank(r.get("실결제금액")) or 0,
            "정산예상금액_배송비포함": settle,
            "마켓수수료": r.get("마켓수수료", ""),
            "수수료율": r.get("수수료율", ""),
            "쇼핑몰": market_to_shopmine(r.get("판매처", "")),
            "수취고객명": r.get("수령자", ""),
            "주문일": r.get("주문일", ""),
            "송장입력": r.get("송장입력", ""),
            "주문상태": r.get("주문상태", ""),
            "_settle_source": src,
            "_sell_origin": "api",
        })
    df = pd.DataFrame(out, columns=SELL_COLUMNS)
    if df.empty:
        df = pd.DataFrame(columns=SELL_COLUMNS)
    return df


def _fetch_rows(since, until, markets):
    """order_export 호출 seam — 테스트에서 monkeypatch 한다.

    한 마켓이라도 실패하면 예외가 전파된다(order_export._fetch_combined 설계).
    부분 성공을 숨기면, 실패한 마켓의 매입 행이 전부 '매출 미매칭'으로 둔갑해
    블랙스팟처럼 보인다 — 조용한 실패보다 나쁜 적극적 오신호. 스펙 §9.
    """
    from lemouton.markets import order_export as oe
    warnings: list = []
    rows = oe.combined_order_rows(markets, since=since, until=until, warnings=warnings)
    return rows, warnings


def from_api(since: _dt.datetime, until: _dt.datetime,
             markets: Optional[list] = None) -> pd.DataFrame:
    """판매처 마켓 API → SellRow DF. df.attrs['warnings'] 에 계정 제외 사유가 담긴다."""
    rows, warnings = _fetch_rows(since, until, markets or API_MARKETS)
    df = _rows_to_df(rows)
    df.attrs["warnings"] = warnings
    logger.info("from_api: rows=%d warnings=%d", len(df), len(warnings))
    return df
```

```python
# -*- coding: utf-8 -*-
"""매입 DF + 매출 DF → 매칭 결과 + 파생 플래그.

원본: C:\\dev\\대량등록 마진계산기\\app.py `_run_full_pipeline` (262행)
차이: supplements(주문번호 보완)는 ② 블랙스팟 사이클로 이월 — 여기서는 다루지 않는다.
"""
import logging
from datetime import datetime

import pandas as pd

from lemouton.margin.buy_parser import split_by_site_order_no
from lemouton.margin.config import DEFAULT_PRICE_RANGES
from lemouton.margin.matcher import match_data

logger = logging.getLogger(__name__)


def _is_empty(v) -> bool:
    return str(v or "").strip() in ("", "nan", "None", "0", "0.0", "NaN")


def _has_real_trace(r: dict) -> bool:
    """진짜 매입 흔적 — 송장번호 / 메모의 URL / 구매가격 > 0."""
    if not _is_empty(r.get("국내송장번호")):
        return True
    if "http" in str(r.get("간단메모", "") or "").lower():
        return True
    try:
        bp = float(str(r.get("구매가격", 0) or 0).replace(",", ""))
        if bp > 0 and bp != 999999999.99:
            return True
    except (ValueError, TypeError):
        pass
    return False


def _parse_date(s):
    s = str(s).strip()
    for fmt in ("%y.%m.%d", "%Y-%m-%d", "%Y.%m.%d"):
        try:
            return datetime.strptime(s[:10] if fmt == "%Y-%m-%d" else s, fmt).date()
        except Exception:  # noqa: BLE001
            continue
    return None


def _classify_price(v, ranges):
    try:
        v = float(v or 0)
    except Exception:  # noqa: BLE001
        v = 0
    for lo, hi, lbl in ranges:
        if lo <= v < hi:
            return lbl
    return "기타"


def run(buy_df: pd.DataFrame, sell_df: pd.DataFrame, price_ranges=None) -> dict:
    """매칭 + 파생 플래그. Returns {matched, unmatched_buy, unmatched_sell, buy_missing}."""
    ranges = price_ranges or DEFAULT_PRICE_RANGES
    buy_valid, buy_missing = split_by_site_order_no(buy_df)

    # 매칭 대상 = buy_valid + buy_missing (G열 없는 행도 마켓주문번호로 매칭 시도)
    full_buy = (pd.concat([buy_valid, buy_missing], ignore_index=True)
                if len(buy_missing) > 0 else buy_valid)
    matched, unmatched_buy, unmatched_sell = match_data(full_buy, sell_df)

    for r in matched:
        if _is_empty(r.get("사이트주문번호")):
            r["_주문미이행"] = True
            if _has_real_trace(r):
                r["_매입흔적"] = True
                r["매칭타입"] = (r.get("매칭타입") or "") + "_매입흔적"

        dt = _parse_date(r.get("주문일", ""))
        r["일자"] = dt.strftime("%Y-%m-%d") if dt else ""
        r["월"] = dt.strftime("%Y-%m") if dt else ""
        r["금액대"] = _classify_price(r.get("판매가"), ranges)

    logger.info("pipeline.run: matched=%d unmatched_buy=%d unmatched_sell=%d",
                len(matched), len(unmatched_buy), len(unmatched_sell))
    return {
        "matched": matched,
        "unmatched_buy": unmatched_buy,
        "unmatched_sell": unmatched_sell,
        "buy_missing": buy_missing.to_dict("records"),
    }
```

```python
# -*- coding: utf-8 -*-
"""마진 분석 세션 저장/조회/정리.

결과는 gzip(JSON) 으로 DB LargeBinary 에. 원본 엑셀은 R2.
최근 KEEP_RECENT 건만 보관 — 초과분은 R2 오브젝트까지 함께 삭제한다.
"""
import gzip
import json
import logging
from typing import Optional

from lemouton.margin.models import MarginAnalysis

logger = logging.getLogger(__name__)

KEEP_RECENT = 20


def _delete_object(key: str) -> None:
    """R2 삭제 seam — 테스트에서 monkeypatch. 실패해도 DB 정리는 진행한다."""
    try:
        from shared import storage
        storage.delete_object(key)
    except Exception:  # noqa: BLE001
        logger.warning("R2 오브젝트 삭제 실패(무시): %s", key, exc_info=True)


def _pack(payload: dict) -> bytes:
    return gzip.compress(json.dumps(payload, ensure_ascii=False).encode("utf-8"))


def _unpack(blob: bytes) -> dict:
    return json.loads(gzip.decompress(blob).decode("utf-8"))


def save(session, *, payload: dict, period_from, period_to,
         buy_file_key: str, buy_filename: str,
         markets_fetched: list, markets_failed: list, counts: dict,
         created_by: Optional[str],
         shopmine_file_key: Optional[str] = None,
         shopmine_filename: Optional[str] = None) -> MarginAnalysis:
    row = MarginAnalysis(
        created_by=created_by,
        period_from=period_from, period_to=period_to,
        buy_file_key=buy_file_key, buy_filename=buy_filename,
        shopmine_file_key=shopmine_file_key, shopmine_filename=shopmine_filename,
        markets_fetched=markets_fetched, markets_failed=markets_failed,
        counts=counts, result_blob=_pack(payload),
    )
    session.add(row)
    session.commit()
    _prune(session)
    return row


def load(session, analysis_id: int) -> dict:
    row = session.get(MarginAnalysis, analysis_id)
    if row is None:
        raise LookupError(f"분석 기록을 찾을 수 없습니다: id={analysis_id}")
    return _unpack(row.result_blob)


def get(session, analysis_id: int) -> Optional[MarginAnalysis]:
    return session.get(MarginAnalysis, analysis_id)


def list_recent(session, limit: int = KEEP_RECENT) -> list:
    return (session.query(MarginAnalysis)
            .order_by(MarginAnalysis.id.desc())
            .limit(limit).all())


def delete(session, analysis_id: int) -> None:
    row = session.get(MarginAnalysis, analysis_id)
    if row is None:
        return
    for key in (row.buy_file_key, row.shopmine_file_key):
        if key:
            _delete_object(key)
    session.delete(row)
    session.commit()


def _prune(session) -> None:
    """KEEP_RECENT 초과분을 오래된 것부터 삭제 (R2 오브젝트 포함)."""
    stale = (session.query(MarginAnalysis)
             .order_by(MarginAnalysis.id.desc())
             .offset(KEEP_RECENT).all())
    for row in stale:
        for key in (row.buy_file_key, row.shopmine_file_key):
            if key:
                _delete_object(key)
        session.delete(row)
    if stale:
        session.commit()
        logger.info("margin store prune: %d건 삭제", len(stale))
```

```python
# -*- coding: utf-8 -*-
"""분석 결과 → xlsx 바이트.

원본: C:\\dev\\대량등록 마진계산기\\app.py `/api/download` (약 1990~2020행)
tab='all' 이면 전체 시트, 특정 탭이면 그 시트만. tab='detail_filtered' 는
프론트가 화면에서 필터한 행과 컬럼 순서를 그대로 받아 쓴다.
"""
import io
from typing import Optional

import pandas as pd

# 탭 키 → (시트명, payload 키)
_SHEETS = [
    ("matched", "전체매칭", "matched"),
    ("unmatched_buy", "마켓X_매입O", "unmatched_buy"),
    ("unmatched_sell", "마켓O_매입X", "unmatched_sell"),
    ("market", "마켓별", "market"),
    ("daily", "일별", "daily"),
    ("monthly", "월별", "monthly"),
    ("brand", "브랜드별", "brand"),
    ("priceRange", "금액대별", "priceRange"),
    ("product", "상품별", "product"),
]


def to_xlsx(payload: dict, tab: str = "all",
            rows: Optional[list] = None,
            column_order: Optional[list] = None) -> bytes:
    buf = io.BytesIO()
    with pd.ExcelWriter(buf, engine="openpyxl") as writer:
        wrote = False

        if tab == "detail_filtered":
            df = pd.DataFrame(rows or [])
            if column_order:
                df = df.reindex(columns=column_order)
            df.to_excel(writer, sheet_name="전체내역", index=False)
            wrote = True
        else:
            if tab in ("all", "summary") and payload.get("summary"):
                (pd.DataFrame([payload["summary"]]).T.reset_index()
                 .rename(columns={"index": "항목", 0: "값"})
                 .to_excel(writer, sheet_name="요약", index=False))
                wrote = True
            for key, sheet, pkey in _SHEETS:
                if tab not in ("all", key):
                    continue
                data = payload.get(pkey) or []
                if not data:
                    continue
                pd.DataFrame(data).to_excel(writer, sheet_name=sheet, index=False)
                wrote = True

        if not wrote:                       # 빈 워크북은 openpyxl 이 못 읽는다
            pd.DataFrame([{"결과": "데이터 없음"}]).to_excel(
                writer, sheet_name="빈결과", index=False)

    return buf.getvalue()
```

```python
# -*- coding: utf-8 -*-
"""마진 계산기 API — /api/margin/*

흐름: upload(기간 추론) → [upload-shopmine] → analyze(API 조회+매칭+저장) → export

마켓 1개라도 조회 실패하면 502 로 중단하고 저장하지 않는다. 실패한 마켓의 매입 행이
전부 '매출 미매칭'으로 둔갑해 블랙스팟처럼 보이기 때문(스펙 §9).
"""
import datetime as _dt
import io
import logging
import uuid

import pandas as pd
from flask import Blueprint, jsonify, request, send_file

from lemouton.margin import aggregator, buy_parser, export, pipeline, sell_source, store
from lemouton.margin.config import DEFAULT_PRICE_RANGES
from shared.db import SessionLocal

logger = logging.getLogger(__name__)

bp = Blueprint("api_margin", __name__, url_prefix="/api/margin")

PERIOD_MARGIN_DAYS = 3
"""마켓 주문일과 더망고 기록일의 어긋남을 흡수. 골든테스트 2단계에서 실데이터로 조정."""

# 업로드 원본 — 요청 간 유지(분석 시 재사용). 프로세스 메모리, 팀 공유 아님.
_PENDING: dict = {}


def _put_object(data: bytes, key: str, content_type: str) -> str:
    """R2 업로드 seam — 테스트에서 monkeypatch."""
    from shared import storage
    storage.put_object(data, key, content_type)
    return key


def _current_user() -> str | None:
    try:
        from flask_login import current_user
        return getattr(current_user, "username", None)
    except Exception:  # noqa: BLE001
        return None


def _infer_period(buy_df: pd.DataFrame):
    dates = [pipeline._parse_date(v) for v in buy_df.get("마켓주문일자", [])]
    dates = [d for d in dates if d]
    if not dates:
        today = _dt.date.today()
        return today - _dt.timedelta(days=7), today
    return (min(dates) - _dt.timedelta(days=PERIOD_MARGIN_DAYS),
            max(dates) + _dt.timedelta(days=PERIOD_MARGIN_DAYS))


@bp.post("/upload")
def upload_buy():
    f = request.files.get("buy_file")
    if f is None:
        return jsonify(error="더망고 엑셀 파일이 없습니다."), 400
    data = f.read()
    try:
        df = buy_parser.parse_buy(data, f.filename)
    except ValueError as e:
        return jsonify(error=str(e)), 400

    since, until = _infer_period(df)
    _PENDING["buy"] = (data, f.filename)
    _PENDING.pop("shopmine", None)
    return jsonify(rows=len(df),
                   markets=sorted(df["마켓명"].dropna().unique().tolist()),
                   period_from=since.isoformat(), period_to=until.isoformat())


@bp.post("/upload-shopmine")
def upload_shopmine():
    """옥션·G마켓 보조 엑셀 (선택). ESM 라이브 검증 완료 시 불필요해진다."""
    f = request.files.get("shopmine_file")
    if f is None:
        return jsonify(error="샵마인 엑셀 파일이 없습니다."), 400
    data = f.read()
    try:
        df = sell_source.from_shopmine_excel(data, f.filename)
    except ValueError as e:
        return jsonify(error=str(e)), 400
    _PENDING["shopmine"] = (data, f.filename)
    return jsonify(rows=len(df),
                   markets=sorted(df["쇼핑몰"].dropna().unique().tolist()))


@bp.post("/analyze")
def analyze():
    if "buy" not in _PENDING:
        return jsonify(error="더망고 엑셀을 먼저 업로드하세요."), 400
    body = request.get_json(silent=True) or {}
    try:
        since = _dt.date.fromisoformat(body["since"])
        until = _dt.date.fromisoformat(body["until"])
    except (KeyError, ValueError):
        return jsonify(error="조회 기간(since·until)이 올바르지 않습니다."), 400

    buy_bytes, buy_name = _PENDING["buy"]
    buy_df = buy_parser.parse_buy(buy_bytes, buy_name)

    kst = _dt.timezone(_dt.timedelta(hours=9))
    try:
        sell_df = sell_source.from_api(
            _dt.datetime.combine(since, _dt.time.min, tzinfo=kst),
            _dt.datetime.combine(until, _dt.time.max, tzinfo=kst))
    except Exception as e:  # noqa: BLE001
        logger.warning("마진 분석 중단 — 마켓 조회 실패", exc_info=True)
        return jsonify(error=f"마켓 주문 조회 실패로 분석을 중단했습니다: {e}"), 502

    warnings = list(sell_df.attrs.get("warnings", []))
    markets_fetched = sell_source.API_MARKETS[:]

    shopmine_key = shopmine_name = None
    if "shopmine" in _PENDING:
        sm_bytes, shopmine_name = _PENDING["shopmine"]
        sm_df = sell_source.from_shopmine_excel(sm_bytes, shopmine_name)
        sell_df = pd.concat([sell_df, sm_df], ignore_index=True)
        markets_fetched.append("shopmine_excel")

    out = pipeline.run(buy_df, sell_df, DEFAULT_PRICE_RANGES)
    agg = aggregator.aggregate(out["matched"], DEFAULT_PRICE_RANGES)
    payload = {**out, **agg}

    buy_key = f"margin/{uuid.uuid4().hex}/{buy_name}"
    _put_object(buy_bytes, buy_key, "application/vnd.ms-excel")
    if shopmine_name:
        shopmine_key = f"margin/{uuid.uuid4().hex}/{shopmine_name}"
        _put_object(_PENDING["shopmine"][0], shopmine_key, "application/vnd.ms-excel")

    counts = {
        "buy": len(buy_df), "sell": len(sell_df),
        "matched": len(out["matched"]),
        "unmatched_buy": len(out["unmatched_buy"]),
        "unmatched_sell": len(out["unmatched_sell"]),
        "settle_estimated": int((sell_df["_settle_source"] == "estimated").sum()),
    }

    session = SessionLocal()
    try:
        row = store.save(session, payload=payload, period_from=since, period_to=until,
                         buy_file_key=buy_key, buy_filename=buy_name,
                         shopmine_file_key=shopmine_key, shopmine_filename=shopmine_name,
                         markets_fetched=markets_fetched, markets_failed=warnings,
                         counts=counts, created_by=_current_user())
        analysis_id = row.id
    finally:
        session.close()

    return jsonify(analysis_id=analysis_id, counts=counts,
                   markets_failed=warnings, **payload)


@bp.get("/analyses")
def list_analyses():
    session = SessionLocal()
    try:
        rows = store.list_recent(session)
        return jsonify(items=[{
            "id": r.id,
            "created_at": r.created_at.isoformat(),
            "created_by": r.created_by,
            "period_from": r.period_from.isoformat(),
            "period_to": r.period_to.isoformat(),
            "buy_filename": r.buy_filename,
            "counts": r.counts,
            "markets_failed": r.markets_failed,
        } for r in rows])
    finally:
        session.close()


@bp.get("/analyses/<int:analysis_id>")
def get_analysis(analysis_id: int):
    session = SessionLocal()
    try:
        payload = store.load(session, analysis_id)
        row = store.get(session, analysis_id)
        return jsonify(analysis_id=analysis_id, counts=row.counts,
                       markets_failed=row.markets_failed, **payload)
    except LookupError as e:
        return jsonify(error=str(e)), 404
    finally:
        session.close()


@bp.delete("/analyses/<int:analysis_id>")
def delete_analysis(analysis_id: int):
    session = SessionLocal()
    try:
        store.delete(session, analysis_id)
        return jsonify(ok=True)
    finally:
        session.close()


@bp.post("/export")
def export_xlsx():
    body = request.get_json(silent=True) or {}
    analysis_id = body.get("analysis_id")
    if not analysis_id:
        return jsonify(error="analysis_id 가 필요합니다."), 400
    session = SessionLocal()
    try:
        payload = store.load(session, int(analysis_id))
    except LookupError as e:
        return jsonify(error=str(e)), 404
    finally:
        session.close()

    data = export.to_xlsx(payload, tab=body.get("tab", "all"),
                          rows=body.get("rows"), column_order=body.get("column_order"))
    return send_file(
        io.BytesIO(data), as_attachment=True,
        download_name=f"마진분석_{analysis_id}.xlsx",
        mimetype="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet")
```

```python
# -*- coding: utf-8 -*-
"""쿠팡 발주서 응답에 '실결제금액' 을 만들 수 있는 필드가 있는지 확인한다.

쿠팡 주문 API 는 실결제금액을 주지 않아, 미정산 주문의 정산 추정을
(단가×수량 + 배송비) × 0.8845 로 계산한다. 샵마인은 실결제금액 × 0.8845 를 쓴다.
쿠폰·즉시할인이 걸리면 값이 갈린다.

사용:  python scripts/margin_spike_coupang_paid.py
출력:  orderItems / shipmentBox 의 전체 키 목록 + 금액 후보 필드
"""
import datetime as _dt
import json

from lemouton.markets.order_export import _account_client
from shared.platforms.coupang.orders import fetch_orders

KST = _dt.timezone(_dt.timedelta(hours=9))

MONEY_HINTS = ("price", "amount", "discount", "paid", "coupon", "Price",
               "Amount", "Discount", "Paid", "Coupon")


def main() -> None:
    client = _account_client("coupang")
    until = _dt.datetime.now(KST)
    since = until - _dt.timedelta(days=7)

    boxes = list(fetch_orders(since, until, client=client))[:5]
    if not boxes:
        print("최근 7일 주문이 없습니다. 기간을 늘려 다시 실행하세요.")
        return

    box_keys, item_keys = set(), set()
    for b in boxes:
        box_keys |= set(b.keys())
        for it in b.get("orderItems", []):
            item_keys |= set(it.keys())

    print("=== shipmentBox 키 ===")
    print(sorted(box_keys))
    print("\n=== orderItems 키 ===")
    print(sorted(item_keys))
    print("\n=== 금액 후보 (box) ===")
    print(sorted(k for k in box_keys if any(h in k for h in MONEY_HINTS)))
    print("\n=== 금액 후보 (item) ===")
    print(sorted(k for k in item_keys if any(h in k for h in MONEY_HINTS)))
    print("\n=== 첫 주문 원본 (금액 필드만) ===")
    b = boxes[0]
    print(json.dumps(
        {k: v for k, v in b.items() if any(h in k for h in MONEY_HINTS)},
        ensure_ascii=False, indent=1))
    for it in b.get("orderItems", [])[:1]:
        print(json.dumps(
            {k: v for k, v in it.items() if any(h in k for h in MONEY_HINTS)},
            ensure_ascii=False, indent=1))


if __name__ == "__main__":
    main()
```

여기서 롯데온 수수료 누락, 쿠팡 추정 오차 같은 **실제 차이가 드러난다.**

`프로그램/_시스템/scripts/margin_api_parity.py`:
```python
# -*- coding: utf-8 -*-
"""골든 테스트 2단계 — 샵마인 엑셀 SellRow vs API SellRow 필드별 대조표.

사용:
    python scripts/margin_api_parity.py "C:\\dev\\대량등록 마진계산기\\데이터\\260704\\20260704_샵마인.xls" 2026-07-01 2026-07-09

출력: 콘솔 요약 + margin_parity_<날짜>.csv
차이를 덮지 않는다. 불일치 상위 건은 판매처 셀러 어드민 화면과 육안 대조한다.
"""
import datetime as _dt
import pathlib
import sys

import pandas as pd

from lemouton.margin import sell_source as SS

KST = _dt.timezone(_dt.timedelta(hours=9))

COMPARE = ["정산예상금액_배송비포함", "단가", "실결제금액", "마켓수수료",
           "수량", "주문상태", "쇼핑몰"]


def main(shopmine_path: str, since_s: str, until_s: str) -> None:
    since = _dt.date.fromisoformat(since_s)
    until = _dt.date.fromisoformat(until_s)

    p = pathlib.Path(shopmine_path)
    sm = SS.from_shopmine_excel(p.read_bytes(), p.name)
    api = SS.from_api(_dt.datetime.combine(since, _dt.time.min, tzinfo=KST),
                      _dt.datetime.combine(until, _dt.time.max, tzinfo=KST))

    for df in (sm, api):
        df["_key"] = df["오픈마켓주문번호"].astype(str).str.strip()

    merged = sm.merge(api, on="_key", how="outer",
                      suffixes=("_샵마인", "_API"), indicator=True)

    only_sm = int((merged["_merge"] == "left_only").sum())
    only_api = int((merged["_merge"] == "right_only").sum())
    both = merged[merged["_merge"] == "both"].copy()

    print(f"주문번호 매칭: 양쪽={len(both)}  샵마인만={only_sm}  API만={only_api}")
    print(f"추정 정산 건수(API): {(api['_settle_source'] == 'estimated').sum()}")
    print(f"정산 없음(API):     {(api['_settle_source'] == 'none').sum()}\n")

    rows = []
    for field in COMPARE:
        a, b = f"{field}_샵마인", f"{field}_API"
        if a not in both or b not in both:
            continue
        diff = both[both[a].astype(str) != both[b].astype(str)]
        print(f"{field:24s} 불일치 {len(diff):5d} / {len(both)}")
        for _, r in diff.iterrows():
            rows.append({"주문번호": r["_key"], "필드": field,
                         "샵마인": r[a], "API": r[b],
                         "쇼핑몰": r.get("쇼핑몰_샵마인", ""),
                         "_settle_source": r.get("_settle_source_API", "")})

    out = pathlib.Path(f"margin_parity_{since_s}_{until_s}.csv")
    pd.DataFrame(rows).to_csv(out, index=False, encoding="utf-8-sig")
    print(f"\n대조표 저장: {out}  ({len(rows)} 행)")

    if rows:
        print("\n=== 쇼핑몰별 불일치 상위 ===")
        print(pd.DataFrame(rows).groupby(["쇼핑몰", "필드"]).size()
              .sort_values(ascending=False).head(15).to_string())


if __name__ == "__main__":
    main(sys.argv[1], sys.argv[2], sys.argv[3])
```
