```python
def list_changes(session, *, set_id, market=None, field=None, limit=200):
    """구성의 변동 이벤트를 최신순(at desc)으로 반환. market/field 선택 필터."""
    q = session.query(ChannelChangeEvent).filter(ChannelChangeEvent.set_id == set_id)
    if market:
        q = q.filter(ChannelChangeEvent.market == market)
    if field:
        q = q.filter(ChannelChangeEvent.field == field)
    q = q.order_by(ChannelChangeEvent.at.desc(), ChannelChangeEvent.id.desc()).limit(limit)
    out = []
    for e in q.all():
        out.append({
            "at": e.at.isoformat() if e.at else None,
            "source": e.source,
            "market": e.market,
            "canonical_sku": e.canonical_sku,
            "field": e.field,
            "prev_value": e.prev_value,
            "next_value": e.next_value,
        })
    return out
```

```javascript
  function histDir(p, n){ if(p==null||n==null||p===n) return ''; return n<p?' class="dn"':' class="up"'; }
  function histArrow(p, n){ if(p==null||n==null||p===n) return ''; return n<p?' ▼':' ▲'; }
  function renderHist(events, fld){
    var src=events.filter(function(e){return e.source==='source';});
    var mkt=events.filter(function(e){return e.source==='market';});
    function col(list){
      if(!list.length) return '<div class="hist-empty">변동 없음</div>';
      return list.map(function(e){
        var d=new Date(e.at); var p=function(n){return(n<10?'0':'')+n;};
        var t=isNaN(d.getTime())?'':(String(d.getFullYear()).slice(2)+'.'+p(d.getMonth()+1)+'.'+p(d.getDate())+' '+p(d.getHours())+':'+p(d.getMinutes()));
        var pv=(e.prev_value==null?'-':fmt(e.prev_value)), nv=(e.next_value==null?'-':fmt(e.next_value));
        return '<div class="hist-ev"><span class="hist-t">'+t+' · '+esc(e.canonical_sku)+'</span>'
          +'<span class="hist-v">'+pv+' → <span'+histDir(e.prev_value,e.next_value)+'>'+nv+histArrow(e.prev_value,e.next_value)+'</span></span></div>';
      }).join('');
    }
    return '<div class="sd-cnt" style="margin-bottom:8px">하락 <span style="color:#1d4ed8;font-weight:700">▼ 파랑</span> · 상승 <span style="color:#dc2626;font-weight:700">▲ 빨강</span> (한국 증시 표기)</div>'
      +'<div class="hist-cols"><div class="hist-col"><div class="hist-h src">소싱처</div>'+col(src)+'</div>'
      +'<div class="hist-col"><div class="hist-h mkt">판매처</div>'+col(mkt)+'</div></div>';
  }
  function openHistModal(sid, mk, fld){
    var nm=mk==='coupang'?'쿠팡':(mk==='smartstore'?'스마트스토어':mk);
    var fn=fld==='price'?'가격':'재고';
    document.getElementById('histTitle').textContent='📈 '+nm+' '+fn+' 변동 이력';
    var body=document.getElementById('histBody'); body.innerHTML='<div class="sd-cnt" style="padding:10px 0">불러오는 중…</div>';
    document.getElementById('histModal').classList.add('on');
    fetch('/api/sets/'+sid+'/history?market='+encodeURIComponent(mk)+'&field='+encodeURIComponent(fld),{headers:{'Accept':'application/json'}})
      .then(function(r){return r.json();})
      .then(function(d){ body.innerHTML=(d&&d.ok)?renderHist(d.events||[], fld):'<div class="hist-empty">불러오지 못했어요.</div>'; })
      .catch(function(){ body.innerHTML='<div class="hist-empty">불러오지 못했어요.</div>'; });
  }
  function closeHistModal(){ document.getElementById('histModal').classList.remove('on'); }
  (function(){ var m=document.getElementById('histModal'); if(!m) return; document.getElementById('histClose').addEventListener('click',closeHistModal); m.addEventListener('click',function(e){ if(e.target.id==='histModal') closeHistModal(); }); })();
```
